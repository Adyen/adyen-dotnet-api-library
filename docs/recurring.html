Recurring

The payment service has the following methods:

listRecurringDetails

Using you can retrieve all cards from a specific shopper. 

var client = new Client();
var recurring = new Service.Recurring(client);

var recurringDetailRequest = new RecurringDetailsRequest
{
  ShopperReference = "test-123",
  MerchantAccount = "merchantAccount",
  Recurring = new PaymentRequest Recurring { Contract = Contract.Oneclick }
};

var recurringDetailResult = new recurring.ListRecurringDetails(recurringDetailRequest);

disable

var client = new Client();
var recurring = new Service.Recurring(client);

var disableRequest = new DisableRequest
{
  ShopperReference = "test-123",
  MerchantAccount = "merchantAccount",
};

var disableResult = new recurring.Disable(disableRequest);
