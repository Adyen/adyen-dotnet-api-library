Directory lookup Hosted Payment Pages

Support additional payment methods using directory lookup. Create a skin on the Adyen side to display these methods. 

// initialize service
var client = new Client();
var hostedPaymentPages = new HostedPaymentPages(client);

var directoryLookupRequest = new DirectoryLookupRequest()
{
    CountryCode = "NL",
    MerchantReference = "Merchant reference",
    PaymentAmount = "1000",
    CurrencyCode = "EUR"
};


var postParameters = hostedPaymentPages.GetPostParametersFromDlRequest(directoryLookupRequest);
